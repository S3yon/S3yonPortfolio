üîç Sample Multi-Table Query

SELECT s.student_id, s.first_name, s.last_name, p.program_name, AVG(g.grade_points) as gpa, COUNT(e.course_id) as courses_taken FROM students s JOIN programs p ON s.program_id = p.program_id JOIN enrollments e ON s.student_id = e.student_id JOIN grades g ON e.enrollment_id = g.enrollment_id WHERE s.enrollment_status = 'ACTIVE' GROUP BY s.student_id, s.first_name, s.last_name, p.program_name HAVING AVG(g.grade_points) >= 3.0 ORDER BY gpa DESC;
